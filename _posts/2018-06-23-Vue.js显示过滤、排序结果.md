```html
  <div id='app'>
    <li v-for='n in filterNumber'> {{ n }} </li>
  </div>
```

```javascript

  var vm = new Vue({
    el: '#app',
    data: {
      numbers: [ 23, 45, 66, 99, 234, 2345 ]
    },
    
    computed: {
      filterNumber: function () {
        return this.numbers.filter(function(numb){
          return numb % 3 === 0
        })
      } 
    }
  })

```
